def isInteger(s):
  s=s.strip()
  if (s[0]=="+" or s[0]=="-") and s[1:].isdigit():
    return True
  if s.isdigit():
    return True
  return False

def main():
  s=input("Vendos nje string: ")
  if isInteger(s):
    print("String paraqet nje integer")
  else:
    print("String nuk paraqet nje integer")

if __name__=="__main__":
  main()
