def reverseLookup(data,value):
  keys=[]
  for key in data:
    if data[key]==value:
      keys.append(key)
  return keys

  FrEn={"le":"the","la":"the","livre":"book","pomme":"apple"}

  print("Expected:['le','la']")
  print()
